import { serve } from "https://deno.land/std@0.168.0/http/server.ts" const corsHeaders = { 'Access-Control-Allow-Origin': '*', 'Access-Control-Allow-Headers': 'authorization, x-client-info, apikey, content-type', } serve(async (req) => { if (req.method === 'OPTIONS') { return new Response('ok', { headers: corsHeaders }) } try { const body = await req.json().catch(() => ({})) const cartItems = body.cartItems if (!cartItems || !Array.isArray(cartItems) || cartItems.length === 0) { throw new Error("Cart is empty or invalid.") } const SQUARE_ACCESS_TOKEN = Deno.env.get('square_access_token') if (!SQUARE_ACCESS_TOKEN) { throw new Error("Square access token is missing in Supabase secrets.") } const isSandbox = SQUARE_ACCESS_TOKEN.startsWith('EAAA') const squareUrl = isSandbox ? "https://connect.squareupsandbox.com/v2/online-checkout/payment-links" : "https://connect.squareup.com/v2/online-checkout/payment-links" const totalAmount = cartItems.reduce((sum, item) => { return sum + (Number(item.price_cents || 0) * Number(item.quantity || 1)) }, 0) if (totalAmount <= 0) { throw new Error("Invalid cart total amount.") } const squareResponse = await fetch(squareUrl, { method: 'POST', headers: { 'Square-Version': '2023-10-18', 'Authorization': `Bearer ${SQUARE_ACCESS_TOKEN}`, 'Content-Type': 'application/json' }, body: JSON.stringify({ idempotency_key: crypto.randomUUID(), quick_pay: { name: "Crown Access Order", price_money: { amount: totalAmount, currency: "USD" } }, checkout_options: { redirect_url: "https://tatumtakia-crypto.github.io" } }) }) const squareData = await squareResponse.json() if (!squareResponse.ok) { throw new Error(squareData.errors?.[0]?.detail || "Failed to create Square payment link.") } return new Response( JSON.stringify({ checkoutUrl: squareData.payment_link.url }), { headers: { ...corsHeaders, 'Content-Type': 'application/json' }, status: 200 } ) } catch (error) { return new Response( JSON.stringify({ error: error.message }), { headers: { ...corsHeaders, 'Content-Type': 'application/json' }, status: 200 } ) } })
